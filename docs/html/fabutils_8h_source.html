106 template <typename T>
107 T moveItems(T dest, T src, size_t n)
108 {
109  T pd = dest;
110  T ps = src;
111  if (pd != ps) {
112  if (ps < pd)
113  for (pd += n, ps += n; n--;)
114  *--pd = *--ps;
115  else
116  while (n--)
117  *pd++ = *ps++;
118  }
119  return dest;
120 }

284 template <typename T>
285 class Stack {
286 public:
287  Stack() : m_items(nullptr) { }
288  bool isEmpty() { return m_items == nullptr; }
289  void push(T const & value) {
290  m_items = new StackItem<T>(m_items, value);
291  }
292  T pop() {
293  if (m_items) {
294  StackItem<T> * iptr = m_items;
295  m_items = iptr->next;
296  T r = iptr->item;
297  delete iptr;
298  return r;
299  } else
300  return T();
301  }
302  int count() {
303  int r = 0;
304  for (auto i = m_items; i; i = i->next)
305  ++r;
306  return r;
307  }
308 private:
309  StackItem<T> * m_items;
310 };

317 template <typename ...Params>
318 struct Delegate {
319 
320  template <typename Func>
321  void operator=(Func f) {
322  m_closure = [] (void * func, const Params & ...params) -> void { (*(Func *)func)(params...); };
323  m_func = heap_caps_malloc(sizeof(Func), MALLOC_CAP_32BIT);
324  moveItems<uint32_t*>((uint32_t*)m_func, (uint32_t*)&f, sizeof(Func) / sizeof(uint32_t));
325  }
326 
327  ~Delegate() {
328  heap_caps_free(m_func);
329  }
330 
331  void operator()(const Params & ...params) {
332  if (m_func)
333  m_closure(m_func, params...);
334  }
335 
336 private:
337  void (*m_closure)(void * func, const Params & ...params);
338  void * m_func = nullptr;
339 };
